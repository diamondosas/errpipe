	package main
	
	import (
		"bufio"
		"fmt"
		"os"
		"os/exec"
		"strings"
	)
	
	func main() {
		scanner := bufio.NewScanner(os.Stdin)
		fmt.Println("🟢 Go shell wrapper started. Type commands:")
	
		for {
			fmt.Print("> ")
			if !scanner.Scan() {
				break
			}
	
			input := strings.TrimSpace(scanner.Text())
			if input == "exit" {
				break
			}
	
			// Intercept & inspect the command BEFORE running
			fmt.Printf("[INTERCEPTED CMD]: %q\n", input)
	
			parts := strings.Fields(input)
			cmd := exec.Command(parts[0], parts[1:]...)
			cmd.Stdin = os.Stdin
	
			output, err := cmd.CombinedOutput()
	
			// Inspect output AFTER running
			fmt.Printf("[INTERCEPTED OUTPUT]:\n%s\n", string(output))
	
			if err != nil {
				fmt.Printf("[ERROR]: %v\n", err)
			}
		}
	}
